#include <iostream>
using namespace std;
void display(int A[],int n)
{
  for(int i=0;i<n;i++)
  {
    cout<<A[i]<<"\t";
  }
}
void bubblesort(int A[],int n)
{
 int temp;
 for(int i=0;i<n-1;i++) //for number of pass
 {
   for(int j=0;j<n-1-i;j++) //for comparison in each pass
   {
     if(A[j]>A[j+1])
     {
      temp=A[j];
      A[j]=A[j+1];
      A[j+1]=temp;
     }
   }
 }
}
int main()
{
 int A[]={5,1,3,2,4};
 int n=5;
 cout<<"INPUT LIST IS: \n";
 display(A,n);
 cout<<"\n";

 bubblesort(A,n);
 cout<<"SORTED LIST IS: \n";
 display(A,n);
 return(0);
}
